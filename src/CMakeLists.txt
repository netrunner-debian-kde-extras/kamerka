SET(kamerka_SRCS 
  main.cpp
  mainwindow.cpp
  videowidget.cpp
  capturethread.cpp
  imageeffect.cpp
  kamerka.qml
  Button.qml
  settingsdialog.cpp
  kamerka.kcfg
)

install(FILES ${kamerka_KCFGS} DESTINATION ${KCFG_INSTALL_DIR})
